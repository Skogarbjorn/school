(define (remove-last x)

      '()
      (cons (car x) (remove-last (cdr x)))
      )

(define (product x)
  (define (producthelp x y)
    (if (null? x)
        y
        (producthelp (cdr x) (* y (car x)))
        )
    )
  (producthelp x 1)
  )
(product '(1 2 3))
(product '())
